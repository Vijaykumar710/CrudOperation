			response.setId(1l);
			response.setName("vijay");
			response.setPassportNumber("A101");
			response.setFilename(fileName);
			response.setSize(size);
			response.setDownloadUri("/downloadFile");
			 
			 return new ResponseEntity<>(response,HttpStatus.OK);
		 }
